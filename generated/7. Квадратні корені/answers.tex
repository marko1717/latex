\textbf{Ключові формули:}

\begin{enumerate}
\item $\sqrt{a^2} = |a|$ --- корінь з квадрата є модулем

\item $|a - b| = \begin{cases} a - b, & \text{якщо } a \geq b \\ b - a, & \text{якщо } a < b \end{cases}$

\item $\sqrt{a} \cdot \sqrt{b} = \sqrt{ab}$, де $a, b \geq 0$

\item $(a - b)(a + b) = a^2 - b^2$ --- різниця квадратів

\item $a^{\frac{m}{n}} = \sqrt[n]{a^m} = \left(\sqrt[n]{a}\right)^m$

\item $a^{-\frac{m}{n}} = \dfrac{1}{a^{\frac{m}{n}}}$

\item $\sqrt[n]{a} \cdot \sqrt[n]{b} = \sqrt[n]{ab}$
\end{enumerate}
